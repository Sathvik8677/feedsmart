function sendOTP(){ let email = document.getElementById("email").value; let formData = new FormData(); formData.append("email", email); fetch('/forgot_send_otp',{ method:'POST', body: formData }) .then(res=>res.json()) .then(d=>{ alert(d.success ? "OTP Sent 📩" : "Email not found ❌"); }); } function verifyOTP(){ let otp = document.getElementById("otp").value; let formData = new FormData(); formData.append("otp", otp); fetch('/forgot_verify_otp',{ method:'POST', body: formData }) .then(r=>r.json()) .then(d=>{ alert(d.success ? "OTP Verified ✅" : "Wrong/Expired ❌"); }); } function resetPassword(){ let password = document.getElementById("new_password").value; let formData = new FormData(); formData.append("password", password); fetch('/reset_password',{ method:'POST', body: formData }) .then(r=>r.json()) .then(d=>{ if(d.success){ alert("Password Reset Successful 🎉"); window.location.href = "/login"; } else { alert("Error ❌"); } }); }
